# 基础操作

#设置初试信息
git config --global user.name "yourname"
git config --global user.email "your_email@youremail.com"

#设置本地秘钥
ssh-keygen -t rsa -C "your_email@youremail.com"

#找到本地生成的文件 C:\Users\Administrator\.ssh
#在云端添加公钥

#将本地目录变成git库
git init

#将本地库添加到远程仓库
git remote add origin https://github.com/yourName/repositoryname.git

#查看工作目录的状态
git status　　　　　　　　　　

#将文件添加到暂存区
git add <file>　　　　　　

#提交更改,添加备注信息(此时将暂存区的信息提交到本地仓库)
git commit -m "commnet" 　　

#将本地仓库的文件push到远程仓库(若 push 不成功，可加 -f 进行强推操作)
git push origin master 　

#从远程仓库pull文件 拉取
git pull origin master 

#创建分支login
git branch login
#切换到master分支,当你切换分支的时候，Git 会用该分支的最后提交的快照替换你的工作目录的内容， 所以多个分支不需要多个目录
git checkout master 

#查看当前分支
git branch 

# 合并分支,将login分支合并到主分支去
git merge login

#推送到云端master分支
git push origin master 


git checkout login
#将本地login分支推送到云端login
git push -u orgin login 


#删除分支
git branch -d login

#删除远程源
git remote rm origin
