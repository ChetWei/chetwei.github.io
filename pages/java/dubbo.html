实现接口

/**
 * Created on 2021/6/12.
 *这里实现了dubbo的接口IAccountDubboApi，作为dubbo服务的提供方
 * @author thomas
 */
@DubboService //使用的是dubbo的注解
public class AccountDubboApiImpl implements IAccountDubboApi {

    @Autowired
    private AccountMapper accountMapper;

    @Override
    public void deductMoney(int userId, BigDecimal money) {

        AccountEntity userDB = accountMapper.selectOne(
                new QueryWrapper<AccountEntity>().lambda().eq(AccountEntity::getUserId, userId));

        userDB.setMoney(userDB.getMoney().subtract(money));

        accountMapper.updateById(userDB);
    }
}

server:
  port: 9001

spring:
  main:
    allow-bean-definition-overriding: true
  application:
    name: app-account-service
  cloud:
    nacos:
      discovery:
        server-addr: localhost:8848
        namespace: bb34df57-b423-4fc0-8dd3-f43dbefc0282
  ######################mysql数据源##########################
  datasource:
    druid:
      url: jdbc:mysql://localhost:3306/fastcloud?serverTimezone=Asia/Shanghai&useUnicode=true&characterEncoding=utf8&rewriteBatchedStatements=true&useSSL=false
      driver-class-name: com.mysql.cj.jdbc.Driver
      username: root
      password: opportunity

#dubbo配置
dubbo:
  application:
    name: ${spring.application.name}
    # qos=Quality of Service 是Dubbo的在线运维命令，可以对服务进行动态的配置、控制(上下线)及查询，具体参考：https://blog.csdn.net/u012988901/article/details/84503672
    qos-enable: false
  #dubbo服务扫描 实现api接口的 包路径
  scan:
    base-packages: com.it.cloud
  #dubbo服务暴露的协议配置 name协议名称，port协议端口 -1 为自增端口20880 开始
  protocol:
    name: dubbo
    port: -1
  registry:
    address: nacos://${spring.cloud.nacos.discovery.server-addr}
    parameters[namespace]: ${spring.cloud.nacos.discovery.namespace}
  cloud:
    subscribed-services: "" #这里服务提供方不需要订阅其他服务，不然默认订阅所有服务消耗资源

2、配置文件

server:
  port: 9002

spring:
  application:
    name: app-order-service

  cloud:
    nacos:
      discovery:
        server-addr: localhost:8848
        namespace: bb34df57-b423-4fc0-8dd3-f43dbefc0282

  ######################mysql数据源##########################
  datasource:
    druid:
      url: jdbc:mysql://localhost:3306/fastcloud?serverTimezone=Asia/Shanghai&useUnicode=true&characterEncoding=utf8&rewriteBatchedStatements=true&useSSL=false
      driver-class-name: com.mysql.cj.jdbc.Driver
      username: root
      password: opportunity

dubbo:
  #dubbo服务暴露的协议配置 name协议名称，port协议端口 -1 为自增端口 20880 开始
  application:
    name: ${spring.application.name}
    qos-enable: false
  protocol:
    name: dubbo
    port: -1
  registry:
    # 其中子属性 address 的值 "spring-com.it.cloud://localhost"，说明挂载到 Spring Cloud 注册中心
    # 使用nacos://localhost:8848表示挂载到nacos注册中心，这种会在nacos上展示服务名称：providers:暴露接口包名:版本:组名
    address: nacos://${spring.cloud.nacos.discovery.server-addr}
    parameters[namespace]: ${spring.cloud.nacos.discovery.namespace}
  cloud:
    subscribed-services: "app-account-service" //订阅服务提供方
